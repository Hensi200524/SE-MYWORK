1.create a procedure:
updateprocedure
table name: products

DELIMITER //

	CREATE PROCEDURE UPDATEPROC(INOUT myid int)
    
    BEGIN
    	
       UPDATE products SET p_category='Electronics' WHERE id = myid;

        
    END//
    
DELIMITER ;

2.deleteprocedure:

DELIMITER //

	CREATE PROCEDURE DELETEPROC(INOUT myid int)
    
    BEGIN
    	
       DELETE FROM products WHERE id = myid;

        
    END//
    
DELIMITER ;

3.selectprocedure:

DELIMITER //

	CREATE PROCEDURE SELECTPROC(INOUT myid int)
    
    BEGIN
    	
       SELECT *  FROM products WHERE id = myid;
       
    END//
    
DELIMITER ;

4.insertprocedure:

DELIMITER //

	CREATE PROCEDURE INSERTPROC(INOUT myid int)
    
    BEGIN
    	
        INSERT into products(p_name,p_category,p_price) VALUES(myid,'western','clothes','600');
        
    END//
    
DELIMITER ;
